########################################################################
# podman-llama — model download + launch cheatsheet
#
# Two target machines:
#   r9700      AMD Radeon AI PRO R9700   (gfx1201, RDNA4, ~31 GiB dedicated VRAM) -> ROCm
#   strixhalo  Ryzen AI Max+ 395         (gfx1151, Radeon 8060S, unified memory)  -> Vulkan
#
# Downloads go to /data/models/<Model>/ (the default MODELS_DIR).
#
# The run scripts are GENERIC. All tuning is injected from env files:
#   models/<MODEL>.env               -> model-intrinsic only (path, sampling)
#   profiles/<MACHINE>/<MODEL>.env   -> optional GPU-specific fit (ctx/ubatch/KV/offload/split)
#   profiles/<MACHINE>/default.env   -> card / machine defaults (GFX, UMA, ctx/batch, device pinning)
# Invoke as:  MACHINE=<machine> ./runtime/<backend>/run-<backend>.sh <model> [extra args...]
########################################################################


########################################################################
# Radeon AI PRO R9700 (ROCm)
########################################################################

# --- downloads ---
hf download unsloth/Qwen3.6-27B-GGUF --include "*Q6_K*" --local-dir /data/models/Qwen3.6-27B
hf download unsloth/GLM-4.7-Flash-GGUF --include "*UD-Q5_K_XL*" --local-dir /data/models/GLM-4.7-Flash
hf download ggml-org/gpt-oss-20b-GGUF gpt-oss-20b-mxfp4.gguf --local-dir /data/models/gpt-oss-20b
hf download unsloth/Devstral-Small-2-24B-Instruct-2512-GGUF --include "*Q6_K*" --local-dir /data/models/Devstral-Small-2-24B

# --- run (set MACHINE=r9700, or export PODMAN_LLAMA_MACHINE=r9700 once — see below) ---
MACHINE=r9700 ./runtime/rocm/run-rocm.sh qwen3-27b
MACHINE=r9700 ./runtime/rocm/run-rocm.sh glm-4.7-flash
MACHINE=r9700 ./runtime/rocm/run-rocm.sh gpt-oss-20b
MACHINE=r9700 ./runtime/rocm/run-rocm.sh devstral-24b

# The model arg names models/<name>.env. Extra llama-server flags pass through and
# OVERRIDE the model env (last wins):
#   MACHINE=r9700 ./runtime/rocm/run-rocm.sh qwen3-27b --temp 0.7 --top-p 0.8
# Override tuning inline:
#   CTX_SIZE=65536 PORT=9090 MACHINE=r9700 ./runtime/rocm/run-rocm.sh glm-4.7-flash
# Raw .gguf path (no model env, launcher defaults apply):
#   MACHINE=r9700 ./runtime/rocm/run-rocm.sh SomeModel/model.gguf --ctx-size 8192

# --- downloads ---
# UD-Q5_K_M is split into 3 shards under a subfolder; the glob grabs all parts.
hf download unsloth/Qwen3-Coder-Next-GGUF --include "UD-Q5_K_M/*" --local-dir /data/models/Qwen3-Coder-Next

# --- run (MACHINE=strixhalo is the default; shown explicit for clarity) ---
# Native 256K context, all layers on GPU (no --n-cpu-moe on unified memory):
MACHINE=strixhalo ./runtime/vulkan/run-vulkan.sh qwen3-coder-next
# ~1M context via static YaRN x4 (separate config: YaRN degrades short-context
# quality, so only launch this when a job actually needs >256K):
MACHINE=strixhalo ./runtime/vulkan/run-vulkan.sh qwen3-coder-next-1m

# The R9700 models also run here via the Vulkan runtime (download them as above):
MACHINE=strixhalo ./runtime/vulkan/run-vulkan.sh qwen3-27b
MACHINE=strixhalo ./runtime/vulkan/run-vulkan.sh glm-4.7-flash
MACHINE=strixhalo ./runtime/vulkan/run-vulkan.sh gpt-oss-20b


########################################################################
# Dual Radeon AI PRO R9700 (Vulkan, model spanned across both cards)
########################################################################
# Two R9700s (~62 GiB VRAM total) span one model across both cards via
# --split-mode/--tensor-split, all weights in VRAM (no CPU-MoE offload).
# GPU-fit tuning lives in profiles/r9700-dual/<model>.env.

# --- downloads ---
# UD-Q4_K_XL (~49.6 GB) is a SINGLE file at the HF repo root (quants <= ~50 GB
# are unsplit); UD-Q5_K_M is a 3-shard subfolder. UD-Q4_K_XL replaces the old
# legacy Q4_1: same size class, imatrix-based, benchmarks above standard quants.
hf download unsloth/Qwen3-Coder-Next-GGUF --include "Qwen3-Coder-Next-UD-Q4_K_XL.gguf" --local-dir /data/models/Qwen3-Coder-Next/UD-Q4_K_XL
hf download unsloth/Qwen3-Coder-Next-GGUF --include "UD-Q5_K_M/*" --local-dir /data/models/Qwen3-Coder-Next

# --- run ---
# UD-Q4_K_XL (~46 GiB) at native 256K context, ~11 GiB VRAM margin (default):
MACHINE=r9700-dual ./runtime/vulkan/run-vulkan.sh qwen3-coder-next-q4
# UD-Q5_K_M (~55 GiB) at 128K (256K is a ~2 GiB-margin squeeze -- see profile):
MACHINE=r9700-dual ./runtime/vulkan/run-vulkan.sh qwen3-coder-next
# If load OOMs, shrink the compute buffer first, then context:
#   UBATCH=256 MACHINE=r9700-dual ./runtime/vulkan/run-vulkan.sh qwen3-coder-next-q4
# Uneven split (e.g. one card drives a display): TENSOR_SPLIT=0.9,1.1


########################################################################
# Global shell profile (set your machine once)
########################################################################
# Add to ~/.bashrc or ~/.zshrc so you never type MACHINE= again:
#
#   export PODMAN_LLAMA_MACHINE=r9700        # or strixhalo, r9700-dual
#
# Optional global overrides (all fall back to the machine profile if unset):
#   export PODMAN_LLAMA_MODELS_DIR=/data/models
#   export PODMAN_LLAMA_PORT=8080
#   export PODMAN_LLAMA_ROCM_IMAGE=llama-strix-halo:rocm
#   export PODMAN_LLAMA_VULKAN_IMAGE=llama-strix-halo:vulkan
#   export PODMAN_LLAMA_CTX_SIZE=32768
#   export PODMAN_LLAMA_UBATCH=2048
#   export PODMAN_LLAMA_BATCH=8192
#   export PODMAN_LLAMA_THREADS=8
#   export PODMAN_LLAMA_GPU_INDEX=0          # vulkan only, multi-GPU hosts
#
# With the global set, just:  ./runtime/rocm/run-rocm.sh glm-4.7-flash


########################################################################
# Per-model config (models/<name>.env — model-intrinsic: path + sampling)
########################################################################
# model                sampling                                       notes
# -------------------  ---------------------------------------------  --------------------------
# gpt-oss-20b          --temp 1.0 --top-p 1.0 --reasoning-format auto MXFP4+SWA: f16 KV (in model env)
# qwen3-27b            --temp 0.6 --top-p 0.95 --top-k 20 --min-p 0   +presence-penalty 1.0
# glm-4.7-flash        --temp 0.7 --top-p 1.0 --min-p 0.01
# devstral-24b         --temp 0.15 --top-p 1.0                        low temp for coding
# qwen3-coder-next     --temp 1.0 --top-p 0.95 --top-k 40 --min-p .01 80B MoE (UD-Q5_K_M ~55 GiB), rp 1.0
# qwen3-coder-next-q4  same sampling                                  80B MoE (UD-Q4_K_XL ~46 GiB)
# qwen3-coder-next-1m  same sampling + YaRN x4 rope scaling           ~1M ctx; YaRN hurts short ctx
#
# models/<name>.env carries ONLY the weight path + sampling. GPU-specific fit
# (CPU-MoE offload, context, KV quant, multi-GPU split) lives per card in
# profiles/<MACHINE>/<name>.env, e.g.:
#   profiles/strixhalo/qwen3-coder-next.env     -> ctx 256k, all-GPU (no MoE offload)
#   profiles/r9700-dual/qwen3-coder-next-q4.env -> ctx 256k, ubatch 512, q8_0 KV
# On AMD keep CACHE_TYPE_K == CACHE_TYPE_V so the fused flash-attn kernel stays on the fast path.


########################################################################
# Claude Code -> llama-server (Anthropic-compatible /v1/messages)
########################################################################
# llama-server exposes /v1/messages natively; --jinja (set by the launchers)
# enables the GGUF's embedded chat template incl. qwen3_coder tool-call
# format (parser fixed upstream in llama.cpp PR #19765, Feb 20 2026).
#
#   export ANTHROPIC_BASE_URL=http://<server>:8080
#   export ANTHROPIC_AUTH_TOKEN=none          # required to be set; value unused
#   export ANTHROPIC_MODEL=qwen3-coder-next   # matches the --alias in the model env
#   claude
#
# Context-shift note for agentic use: nothing to configure. llama.cpp's
# context shift is default-OFF, and for hybrid (DeltaNet) models the server
# force-disables it regardless (recurrent state can't be shifted). When the
# context fills, the request errors instead of silently dropping the prefix
# -- prefer that over corrupted long sessions.


########################################################################
# Precedence (highest first)
########################################################################
#   trailing CLI args  >  inline env  >  models/<name>.env
#     >  profiles/<MACHINE>/<name>.env  >  profiles/<MACHINE>/default.env  >  built-in default
# (Env files use := "set if unset", so the FIRST assignment of each var wins; the
#  launchers source in precedence order so earlier files pin and later ones fill gaps.)
#
# Run any model on another machine by flipping MACHINE:
#   MACHINE=strixhalo ./runtime/rocm/run-rocm.sh glm-4.7-flash     # ROCm on Strix Halo
#   MACHINE=r9700     ./runtime/vulkan/run-vulkan.sh qwen3-27b     # Vulkan on one R9700
